Step 1

Create a package.json file.
This defines our application and all dependencies.


{ 
    "name": "employee-api",
    "description": "RESTful API to get employee information",
    "version": "0.0.1",
    "dependencies": {
        "express": "3.x",
        "mongoose": "3.8.8",
        "bcrypt": "0.7.7"
    }
}

Step 2

Create a app.js file.


var express = require('express'),
    mongoose = require('mongoose'), 
    bcrypt = require('bcrypt'),
    database = require('./config/database');

// Create app and use JSON/urlencoded parsing middleware
var app = express();
app.use(express.json());
app.use(express.urlencoded());

require('./routes')(app);

mongoose.connect(database.url);

var server = app.listen(5000, function() {
  console.log('Listening on: http://localhost:%d', server.address().port);
});

Step 3

Install Dependencies


npm install

Run Application


node app.js


You are now running a web service on port 5000!

Step 4

Create Models to define our data schema and validation.


var mongoose = require('mongoose'), Schema = mongoose.Schema;

var skillSchema = new Schema({ 
    title: { type: String, required: true },
    rating: { type: Number, max: 5, required: true }
});

var userSchema = new Schema({
    name: { type: String, required: true },
    headshot: { type: String },
    startDate: { type: Date },
    jobTitle: { type: String, required: true },
    email: { type: String },
    skills: [skillSchema]
});

module.exports = mongoose.model('User', userSchema);

Step 5: Define Routes

Routes are URL schema, that describe
the interfaces for making requests to your web app.



Route Examples:

GET /users → returns all users

POST /login {JSON Data} → returns key if valid

GET /users/{userID} → returns one user

app.delete('/users/:userid', checkAuth, function(req, res) {
  
  User.findByIdAndRemove(req.params.userid, function (err, resource) {
    
    if (err) { 
      res.jsonp(500, { error: err.name + ' - ' + err.message });
    } else if (resource) {
      res.jsonp({ message: 'User deleted' });
    } else {
      res.jsonp(404, { error: 'User not found' });
    }
  });
});

Security


We need to do the following:

  • Encrypt passwords
  • Set protected routes

Encrypt Passwords

Hash and salt passwords with bcrypt


node addUser.js bmitchell secretpassword

Save it:


bcrypt.hash('secretpassword', 10, function( err, bcryptedPassword) {
    // Store bcryptedPassword in Mongo DB
});

Compare it:


bcrypt.compare('secretpassword', bcryptedPassword, function(err, match) {
    // match returns true if valid
});

Protected Routes

  • Need to protect routes that modify users
  • POST PUT DELETE commands
  • Administrators: read/write
  • Public: read-only



How do we do this?

API Keys

Require keys when modifying users

POST /login {username/password}
↓
Obtain Key
↓
POST /users?key={valid key} {new user data}

Temporary Keys

Another layer of protection

  • Expire keys after X amount of time
  • Mongo has TTL built in for documents
  • TTL stands for "Time To Live"


Defined at the schema level:


var keySchema = new Schema({
  createdAt: { type: Date, expires: 3600, default: Date.now, unique: true }
});
