Step 1

Create a package.json file.
This defines our application and all dependencies.


{ 
    "name": "employee-api",
    "description": "RESTful API to get employee information",
    "version": "0.0.1",
    "dependencies": {
        "express": "3.x",
        "mongoose": "3.8.8",
        "bcrypt": "0.7.7"
    }
}

Step 2

Create a app.js file.
This creates and launches our application on port 5000


var express = require('express'), mongoose = require('mongoose'), 
    bcrypt = require('bcrypt'), database = require('./config/database');

// Create app and use JSON/urlencoded parsing middleware
var app = express();
app.use(express.json());
app.use(express.urlencoded());

require('./routes')(app);

mongoose.connect(database.url, function(err) { 
    if (err) myConsole('Error: Unable to connect to MongoDB!');
});

var server = app.listen(5000, function() {
    console.log('Listening on: http://localhost:%d', server.address().port);
});

Step 3

Create Models to define our data schema and validation.


var mongoose = require('mongoose'), Schema = mongoose.Schema;

var skillSchema = new Schema({ 
    title: { type: String, required: true },
    rating: { type: Number, max: 5, required: true }
});

var userSchema = new Schema({
    name: { type: String, required: true },
    headshot: { type: String },
    startDate: { type: Date },
    jobTitle: { type: String, required: true },
    headshot: { type: String },
    skills: [skillSchema]
});

module.exports = mongoose.model('User', userSchema);

Define Routes

Routes are URL schema, which describe

Security

Protect against malicious users that try to:

  • Steal Mongo DB data files
  • Read/modify data through protected API functions

Encrypt Passwords

Use bcrypt to hash passwords before storing.


node addUser.js bmitchell secretpassword 3
					


var mongoose = require('mongoose'), bcrypt = require('bcrypt'),
    Login = require('../models/login');

bcrypt.hash(process.argv[3], 10, function( err, bcryptedPassword) {
    var query = { username: process.argv[2] };
    Login.update(query, {$set: { username: process.argv[2].toLowerCase(), 
                 password: bcryptedPassword, level: process.argv[4] }}, 
                 { upsert: true }, function (err, numberAffected, raw) {
        if (err) { 
            console.log('Error: Unable to add user');
        } else {
            console.log('User added: ' + process.argv[2].toLowerCase());

        }
}); });

Temporary Keys

  • Valid key is needed to use protected API functions
  • Keys should expire after a set amount of time
  • Mongo allows us to set a TTL for documents.
  • TTL stands for "Time To Live"
  • If a malicious user obtained a key, it would expire soon
